        const mapping = {

            'അ': 'a', 'ആ': 'aa', 'ഇ': 'i', 'ഈ': 'ee', 'ഉ': 'u', 'ഊ': 'oo', 'ഋ': 'ri', 'എ': 'e', 'ഏ': 'e', 'ഐ': 'ai', 'ഒ': 'o', 'ഓ': 'o', 'ഔ': 'au',

            'ക': 'ka', 'ഖ': 'kha', 'ഗ': 'ga', 'ഘ': 'gha', 'ങ': 'nga',

            'ച': 'cha', 'ഛ': 'chha', 'ജ': 'ja', 'ഝ': 'jha', 'ഞ': 'nja',

            'ട': 'da', 'ഠ': 'tha', 'ഡ': 'da', 'ഢ': 'dha', 'ണ': 'na',

            'ത': 'tha', 'ഥ': 'thha', 'ദ': 'dha', 'ധ': 'dhha', 'ന': 'na',

            'പ': 'pa', 'ഫ': 'pha', 'ബ': 'ba', 'ഭ': 'bha', 'മ': 'ma',

            'യ': 'ya', 'ര': 'ra', 'ല': 'la', 'വ': 'va', 'ശ': 'sha', 'ഷ': 'sha', 'സ': 'sa', 'ഹ': 'ha',

            'ള': 'la', 'ഴ': 'zha', 'റ': 'ra',

            'ാ': 'aa', 'ി': 'i', 'ീ': 'ee', 'ു': 'u', 'ൂ': 'oo', 'ൃ': 'ri',

            'െ': 'e', 'േ': 'e', 'ൈ': 'ai', 'ൊ': 'o', 'ോ': 'o', 'ൗ': 'au',

            'ം': 'm', 'ഃ': 'h',

            'ൻ': 'n', 'ർ': 'r', 'ൽ': 'l', 'ൾ': 'l', 'ൺ': 'n', 'ൿ': 'k',

            'റ്റ': 'tta'

        };


        const vowelSigns = ['ാ', 'ി', 'ീ', 'ു', 'ൂ', 'ൃ', 'െ', 'േ', 'ൈ', 'ൊ', 'ോ', 'ൗ'];

        

        function capitalizeFirstLetter(string) {

            if (!string) return string;

            let str = string.trimStart();

            if (str.length === 0) return string;

            return str.charAt(0).toUpperCase() + str.slice(1);

        }


        function transliterateMalayalam(text) {

            let result = "";

            for (let i = 0; i < text.length; i++) {

                let char = text[i];

                let nextChar = text[i + 1];


                if (char === 'റ' && nextChar === '്' && text[i+2] === 'റ') {

                    result += "tta";

                    i += 2;

                    continue;

                }


                if (mapping[char]) {

                    if (mapping[char].endsWith('a') && char.length === 1 && !'അആഇഈഉഊഋഎഏഐഒഓഔ'.includes(char)) {

                        if (vowelSigns.includes(nextChar)) {

                            result += mapping[char].slice(0, -1) + mapping[nextChar];

                            i++;

                        } else if (nextChar === '്') {

                            result += mapping[char].slice(0, -1);

                            i++;

                        } else {

                            result += mapping[char];

                        }

                    } else {

                        result += mapping[char];

                    }

                } else {

                    result += char;

                }

            }

            

            return result

                .replace(/aa-i/g, 'aayi')

                .replace(/aa-u/g, 'aayu')

                .replace(/aa-e/g, 'aaye')

                .replace(/u-u/g, 'u')

                .replace(/e-e/g, 'e')

                .replace(/dd/g, 'tt')

                .replace(/rra/g, 'tta')

                .replace(/rri/g, 'tti')

                .replace(/nre/g, 'nte')

                .replace(/kripa/gi, 'Kripa');

        }


        function processText() {

            const inputField = document.getElementById('input');

            const input = inputField.value;

            const status = document.getElementById('status');

            

            if (!input.trim()) {

                status.innerText = "Input is empty!";

                return;

            }


            status.innerText = "Processing...";

            

            setTimeout(() => {

                const lines = input.split('\n');

                const processedLines = lines.map(line => {

                    let trimmed = line.trim();

                    

                    if (!trimmed || /^\d+$/.test(trimmed) || trimmed.includes('====')) {

                        return line;

                    }


                    let transliterated = "";


                    if (line.startsWith('**')) {

                        transliterated = transliterateMalayalam(line.substring(2));

                        return '**' + capitalizeFirstLetter(transliterated);

                    }

                    

                    const verseMatch = line.match(/^(\d+\.)\s*(.*)/);

                    if (verseMatch) {

                        transliterated = transliterateMalayalam(verseMatch[2]);

                        return verseMatch[1] + ' ' + capitalizeFirstLetter(transliterated);

                    }


                    if (line.trim().startsWith('(')) {

                        return transliterateMalayalam(line);

                    }


                    transliterated = transliterateMalayalam(line);

                    return capitalizeFirstLetter(transliterated);

                });

                

                document.getElementById('output').value = processedLines.join('\n');

                status.innerText = "Success!";

                setTimeout(() => status.innerText = "", 3000);

            }, 50);

        }


        function copyToClipboard() {

            const output = document.getElementById('output');

            output.select();

            document.execCommand('copy');

            const status = document.getElementById('status');

            status.innerText = "Copied to clipboard!";

            setTimeout(() => status.innerText = "", 3000);

        }
